    private void jButton14ActionPerformed(java.awt.event.ActionEvent evt) {                                          
        // TODO add your handling code here:'
         i.push(Double.parseDouble(jTextField1.getText()));
        jTextField1.setText("");
       
        while(!j.isEmpty()){
            double a= i.pop();
            double b=i.pop();
            
            if(j.peek()=="+"){
                
                i.push(a+b);
                System.out.println("1"+i.peek());
                j.pop();
            }
           else if(j.peek()=="-"){
                 i.push(a-b);
                 j.pop();
                 System.out.println("2"+i.peek());
           }
            else if(j.peek()=="/"){
                i.push(b/a);
                j.pop();
                System.out.println("3"+i.peek());
            }
            else if(j.peek()=="*"){
                i.push(a*b);
                j.pop();
                System.out.println("4"+i.peek());
            }
        }
        
        String z = Double.toString(i.peek());
        i.pop();
        jLabel3.setText(z);
    }
